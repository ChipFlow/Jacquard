4//! GF180MCU cell classification.
5//!
6//! Each query function takes the base cell type (the result of
7//! [`crate::gf180mcu::extract_cell_type`]) — never the full
8//! prefixed/drive-suffixed macro name. Classification is by exact
9//! match per cell type rather than prefix, mirroring the lesson from
10//! `docs/adding-a-pdk.md` § Common Pitfalls (e.g. `dlygate4sd3` vs
11//! latch cells in SKY130).
12//!
13//! Sibling of [`crate::sky130_pdk`]. AIG decomposition rules and
14//! behavioural-model parsing land in Phase 4 per
15//! `docs/plans/gf180mcu-enablement.md`.
16//!
17//! # Reset polarity
18//!
19//! GF180MCU uses **active-low** resets and sets, following the same
20//! convention as SKY130. Reset ports are named `RN` and set ports
21//! `SETN`. The "n" *prefix* in cell names like `dffnq` / `dffnrnq` /
22//! `icgtn` instead indicates a **negative-edge clock** — the relevant
23//! port is then `CLKN` rather than `CLK`.
24
25#![allow(unused)]
26
27/// Flip-flops + latches + clock-gating cells — anything whose Q output
28/// captures state across a clock edge (or transparency window).
29pub fn is_sequential_cell(cell_type: &str) -> bool {
30    matches!(
31        cell_type,
32        // Positive- and negative-edge DFFs (R = active-low reset RN,
33        // S = active-low set SETN, sdff = scan DFF with SE/SI pins).
34        "dffq" | "dffnq"
35            | "dffrnq" | "dffnrnq"
36            | "dffsnq" | "dffnsnq"
37            | "dffrsnq" | "dffnrsnq"
38            | "sdffq" | "sdffrnq" | "sdffsnq" | "sdffrsnq"
39            // Level-sensitive latches (E enable).
40            | "latq" | "latrnq" | "latsnq" | "latrsnq"
41            // Clock-gating cells (icgtp = positive enable, icgtn = negative).
42            | "icgtp" | "icgtn"
43    )
44}
45
46/// Constant-driving tie cells. Their output is always 1 (`tieh`) or 0
47/// (`tiel`) and they have no functional inputs.
48pub fn is_tie_cell(cell_type: &str) -> bool {
49    matches!(cell_type, "tieh" | "tiel")
50}
51
52/// Physical-only cells: fillers, decap, end-cap, antenna diodes.
53/// Recognised so post-P&R netlists parse, but they contribute no
54/// logic to the AIG.
55pub fn is_filler_cell(cell_type: &str) -> bool {
56    matches!(
57        cell_type,
58        "antenna" | "endcap" | "fill" | "fillcap" | "filltie"
59    )
60}
61
62/// Hold-time repair buffers and explicit delay cells — combinational,
63/// I → Z, but inserted by P&R for timing reasons rather than logic.
64pub fn is_delay_cell(cell_type: &str) -> bool {
65    matches!(cell_type, "dlya" | "dlyb" | "dlyc" | "dlyd" | "hold")
66}
67
68/// Cells with more than one functional output. The AIG builder
69/// processes one output pin at a time, so multi-output cells need
70/// per-output decomposition rules in Phase 4.
71pub fn is_multi_output_cell(cell_type: &str) -> bool {
72    // Full adder (S, CO) and half adder (S, CO). All other gf180mcu
73    // standard cells are single-output.
74    matches!(cell_type, "addf" | "addh")
75}

77#[cfg(test)]
78mod tests {
79    use super::*;
80
81    /// The full GF180MCU base cell-type catalogue. Both the 7t5v0 and
82    /// 9t5v0 standard cell libraries ship these 69 cells; the lists
83    /// are identical (verified in build.rs's dedup pass).
84    const ALL_CELL_TYPES: &[&str] = &[
85        "addf", "addh", "and2", "and3", "and4", "antenna", "aoi21", "aoi211", "aoi22",
86        "aoi221", "aoi222", "buf", "bufz", "clkbuf", "clkinv", "dffnq", "dffnrnq",
87        "dffnrsnq", "dffnsnq", "dffq", "dffrnq", "dffrsnq", "dffsnq", "dlya", "dlyb",
88        "dlyc", "dlyd", "endcap", "fill", "fillcap", "filltie", "hold", "icgtn", "icgtp",
89        "inv", "invz", "latq", "latrnq", "latrsnq", "latsnq", "mux2", "mux4", "nand2",
90        "nand3", "nand4", "nor2", "nor3", "nor4", "oai21", "oai211", "oai22", "oai221",
91        "oai222", "oai31", "oai32", "oai33", "or2", "or3", "or4", "sdffq", "sdffrnq",
92        "sdffrsnq", "sdffsnq", "tieh", "tiel", "xnor2", "xnor3", "xor2", "xor3",
93    ];
94
95    #[test]
96    fn catalogue_size_matches_pin_table() {
97        assert_eq!(ALL_CELL_TYPES.len(), 69);
98    }
99
100    #[test]
101    fn sequential_classification_is_correct() {
102        // DFFs + latches + clock gating = 12 + 4 + 2 = 18 cells.
103        let seq: Vec<&&str> = ALL_CELL_TYPES
104            .iter()
105            .filter(|c| is_sequential_cell(c))
106            .collect();
107        assert_eq!(
108            seq.len(),
109            18,
110            "expected 18 sequential cells, got {:?}",
111            seq
112        );
113    }
114
115    #[test]
116    fn tie_classification() {
117        assert!(is_tie_cell("tieh"));
118        assert!(is_tie_cell("tiel"));
119        assert!(!is_tie_cell("dffq"));
120        assert!(!is_tie_cell("inv"));
121    }
122
123    #[test]
124    fn filler_classification() {
125        assert!(is_filler_cell("fill"));
126        assert!(is_filler_cell("fillcap"));
127        assert!(is_filler_cell("filltie"));
128        assert!(is_filler_cell("endcap"));
129        assert!(is_filler_cell("antenna"));
130        assert!(!is_filler_cell("hold"));
131        assert!(!is_filler_cell("buf"));
132    }
133
134    #[test]
135    fn delay_classification() {
136        for c in ["dlya", "dlyb", "dlyc", "dlyd", "hold"] {
137            assert!(is_delay_cell(c), "{c} should be a delay cell");
138        }
139        // Plain buffers aren't delay-repair cells.
140        assert!(!is_delay_cell("buf"));
141        assert!(!is_delay_cell("clkbuf"));
142    }
143
144    #[test]
145    fn multi_output_classification() {
146        assert!(is_multi_output_cell("addf"));
147        assert!(is_multi_output_cell("addh"));
148        // DFFs only emit Q (no QN counterpart in the gf180mcu catalogue).
149        assert!(!is_multi_output_cell("dffq"));
150        assert!(!is_multi_output_cell("dffrnq"));
151        // Inverters are single-output.
152        assert!(!is_multi_output_cell("inv"));
153    }
154
155    #[test]
156    fn categories_partition_the_catalogue_disjointly() {
157        // Every cell belongs to at most one of {sequential, tie, filler,
158        // delay}. Multi-output is orthogonal (an adder is combinational
159        // but multi-output). Catches misclassification regressions.
160        for c in ALL_CELL_TYPES {
161            let memberships = [
162                is_sequential_cell(c),
163                is_tie_cell(c),
164                is_filler_cell(c),
165                is_delay_cell(c),
166            ];
167            let count = memberships.iter().filter(|&&x| x).count();
168            assert!(
169                count <= 1,
170                "cell {c} matches more than one classification: {:?}",
171                memberships
172            );
173        }
174    }
175
176    #[test]
177    fn combinational_cells_have_no_classification() {
178        // Pure combinational gates (inv, nand2, oai21, etc.) should NOT
179        // match any specialised classifier.
180        for c in [
181            "inv", "buf", "nand2", "nor3", "and4", "or2", "xor2", "xnor3", "aoi22",
182            "oai222", "mux2", "mux4",
183        ] {
184            assert!(!is_sequential_cell(c), "{c} should not be sequential");
185            assert!(!is_tie_cell(c), "{c} should not be a tie cell");
186            assert!(!is_filler_cell(c), "{c} should not be a filler");
187            assert!(!is_delay_cell(c), "{c} should not be a delay cell");
188        }
189    }
