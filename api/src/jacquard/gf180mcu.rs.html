4//! GF180MCU standard-cell library support.
5//!
6//! Sibling of [`crate::sky130`]; together with [`crate::gf180mcu_pdk`] this
7//! module provides cell detection, pin direction, and library identification
8//! for GlobalFoundries' open-source 180 nm MCU PDK across both
9//! `gf180mcu_fd_sc_mcu7t5v0` (7-track, 5 V) and `gf180mcu_fd_sc_mcu9t5v0`
10//! (9-track, 5 V) standard-cell libraries.
11//!
12//! Cell-name convention:
13//! `gf180mcu_fd_sc_<track>__<celltype>_<drive>`
14//! e.g. `gf180mcu_fd_sc_mcu7t5v0__nand2_1`.
15//!
16//! IO pads (`gf180mcu_fd_io__*`), primitives (`gf180mcu_fd_pr__*`), and IP
17//! blocks (`gf180mcu_fd_ip_*__*`) all share the `gf180mcu_fd_` prefix and
18//! are detected as GF180MCU here so the netlist parser doesn't reject them;
19//! cell-type extraction is conservative and only strips the standard-cell
20//! prefixes. Decomposition for non-standard-cell families lands in later
21//! phases per `docs/plans/gf180mcu-enablement.md`.
22
23/// All GF180MCU cell instantiations share this prefix.
24const GF180MCU_PREFIX: &str = "gf180mcu_fd_";
25
26const SC_7T_PREFIX: &str = "gf180mcu_fd_sc_mcu7t5v0__";
27const SC_9T_PREFIX: &str = "gf180mcu_fd_sc_mcu9t5v0__";
28
29/// Is `name` any GF180MCU cell — standard cell, IO pad, primitive, or IP?
30pub fn is_gf180mcu_cell(name: &str) -> bool {
31    name.starts_with(GF180MCU_PREFIX)
32}
33
34/// Strip the GF180MCU library prefix and the trailing drive-strength
35/// suffix to recover the base cell type.
36///
37/// `gf180mcu_fd_sc_mcu7t5v0__nand2_1` → `nand2`.
38/// `gf180mcu_fd_sc_mcu9t5v0__inv_20` → `inv`.
39///
40/// Non-standard-cell families (IO, PR, IP) return the suffix after the
41/// library prefix verbatim — decomposition for those lands in later
42/// phases.
43pub fn extract_cell_type(name: &str) -> &str {
44    let stripped = name
45        .strip_prefix(SC_7T_PREFIX)
46        .or_else(|| name.strip_prefix(SC_9T_PREFIX))
47        .unwrap_or(name);
48
49    // Drive strength is always a numeric tail after the final underscore
50    // — matches the SKY130 convention.
51    if let Some(last_underscore) = stripped.rfind('_') {
52        let suffix = &stripped[last_underscore + 1..];
53        if !suffix.is_empty() && suffix.chars().all(|c| c.is_ascii_digit()) {
54            return &stripped[..last_underscore];
55        }
56    }
57
58    stripped
59}

61#[cfg(test)]
62mod tests {
63    use super::*;
64
65    #[test]
66    fn detects_standard_cells_in_both_variants() {
67        assert!(is_gf180mcu_cell("gf180mcu_fd_sc_mcu7t5v0__nand2_1"));
68        assert!(is_gf180mcu_cell("gf180mcu_fd_sc_mcu9t5v0__nand2_1"));
69    }
70
71    #[test]
72    fn detects_io_pr_and_ip_families() {
73        assert!(is_gf180mcu_cell("gf180mcu_fd_io__bi_24t"));
74        assert!(is_gf180mcu_cell("gf180mcu_fd_pr__nmos_3p3"));
75        assert!(is_gf180mcu_cell("gf180mcu_fd_ip_sram__sram64x16m8wm1"));
76    }
77
78    #[test]
79    fn rejects_non_gf180mcu_names() {
80        assert!(!is_gf180mcu_cell("sky130_fd_sc_hd__nand2_1"));
81        assert!(!is_gf180mcu_cell("AND2_00_0"));
82        assert!(!is_gf180mcu_cell(""));
83        assert!(!is_gf180mcu_cell("gf180mcu"));
84    }
85
86    #[test]
87    fn extracts_base_cell_type_across_variants() {
88        assert_eq!(extract_cell_type("gf180mcu_fd_sc_mcu7t5v0__inv_1"), "inv");
89        assert_eq!(extract_cell_type("gf180mcu_fd_sc_mcu7t5v0__inv_20"), "inv");
90        assert_eq!(extract_cell_type("gf180mcu_fd_sc_mcu9t5v0__nand2_4"), "nand2");
91        assert_eq!(extract_cell_type("gf180mcu_fd_sc_mcu9t5v0__dffq_1"), "dffq");
92        assert_eq!(
93            extract_cell_type("gf180mcu_fd_sc_mcu7t5v0__aoi222_1"),
94            "aoi222"
95        );
96    }
97
98    #[test]
99    fn extraction_handles_dff_with_polarity_in_name() {
100        // The 'n' in dffnq / dffnrnq is part of the cell type, not a
101        // drive-strength suffix — the drive number is the trailing _1.
102        assert_eq!(extract_cell_type("gf180mcu_fd_sc_mcu7t5v0__dffnq_1"), "dffnq");
103        assert_eq!(extract_cell_type("gf180mcu_fd_sc_mcu7t5v0__dffnrsnq_2"), "dffnrsnq");
104    }
105
106    #[test]
107    fn extraction_passes_through_unprefixed_names() {
108        // Unprefixed names still get numeric-suffix stripping.
109        assert_eq!(extract_cell_type("inv_1"), "inv");
110        // Non-numeric tails are left intact — no false trim.
111        assert_eq!(extract_cell_type("not_a_cell"), "not_a_cell");
112        assert_eq!(extract_cell_type("plain"), "plain");
113    }
114}
