4//! PDK-neutral behavioral parsing and AIG decomposition primitives.
5//!
6//! The `.functional.v` cell-model files shipped by open-source PDKs
7//! (SKY130, GF180MCU, and others that follow the same OpenLane /
8//! OpenROAD ecosystem conventions) use a small fixed grammar:
9//!
10//! - One `module <name>( <ports> );` declaration.
11//! - `input`/`output` port-direction lines.
12//! - A topologically ordered sequence of Verilog gate-primitive
13//!   instantiations (`not`, `buf`, `and`, `or`, `nand`, `nor`,
14//!   `xor`, `xnor`) plus optional UDP instantiations whose names
15//!   are PDK-specific (e.g. `sky130_fd_sc_hd__udp_*`,
16//!   `UDP_GF018hv5v_mcu_sc7_*`).
17//!
18//! The parser is fully prefix-agnostic; UDP entries are surfaced
19//! verbatim and the PDK-specific decomposition layer routes them
20//! to PDK-specific UDP handlers.
21//!
22//! Per the GF180MCU enablement plan (Phase 4-pre), this module
23//! exists to make the PDK-neutral surface explicit, so GF180MCU
24//! (and future PDKs) can call into it without depending on the
25//! sky130_pdk module name. The underlying definitions still live
26//! in [`crate::sky130_pdk`] — physical relocation is deferred to
27//! a follow-up cleanup pass.
28
29pub use crate::sky130_pdk::{
30    parse_functional_model, parse_udp, BehavioralGate, BehavioralModel, UdpModel, UdpRow,
31};
32
33#[cfg(test)]
34mod tests {
35    use super::*;
36
37    /// Sentinel: a regression catching any future change that breaks
38    /// PDK-neutrality of the behavioural parser. Mirrors the recon
39    /// tests in `gf180mcu_pdk::tests` but goes through the neutral
40    /// re-export path callers are expected to use.
41    #[test]
42    fn parser_reachable_via_neutral_module() {
43        let src = "module tiny( A, Y );\ninput A;\noutput Y;\n\tnot u(Y, A);\nendmodule\n";
44        let m: BehavioralModel = parse_functional_model(src).expect("parse");
45        assert_eq!(m.module_name, "tiny");
46        assert_eq!(m.gates.len(), 1);
47        assert_eq!(m.gates[0].gate_type, "not");
48    }
49}
